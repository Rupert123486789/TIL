N, K = map(int, input().split())

a = 0
b = 0

arr = list(map(int, input().split()))

ans = 0

for n in arr:
    if a + b == K:
        ans += 1
        if a > 0:
            a -= 1
        if b > 0:
            b -= 1
    if n == 1:
        a += 1
    else:
        b += 1


m = min(a, b)
ans += m
a -= m
b -= m

ans += a
ans += b

print(ans)
