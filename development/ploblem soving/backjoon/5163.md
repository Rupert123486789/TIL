```python
import sys
import math

input = sys.stdin.readline

T = int(input())

cnt = 0
for _ in range(T):
    ans = 0
    b, w = map(float, input().split())
    cnt += 1
    for _ in range(int(b)):
        r = float(input())
        ans += (4/3) * math.pi * (r**3)

    if ans/1000 >= w:
        print(f'Data Set {cnt}:')
        print('Yes')
    else:
        print(f'Data Set {cnt}:')
        print('No')
    print()
```
