# Animal Farm

```python
import sys
input = sys.stdin.readline

N = int(input())

lst = []
ans = 0
for _ in range(N):
    A, B = input().split()
    if A  == 'pig':
        ans = max(ans, int(B))
    else:
        lst.append(int(B))

res = ans
for i in lst:
    if i <ans:
        res += i

print(res)
```
