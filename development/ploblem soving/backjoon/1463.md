# 1로 만들기

```python
N = int(input())

D = [0] * (N+1)
for i in range(2, N+1):                 # 1을 만들어야 되므로 2부터 시작
    D[i] = D[i-1] + 1                   # -1일때 횟수 1증가
    if i%2 == 0:                        # 2로 나눌 수 있으면
        D[i] = min(D[i], D[i//2]+1)     # 더 적은 것으로 갱신(i//2에서 2를 곱해야 i 이므로 횟수 1증가)
    if i%3 == 0:                        # 3으로 나눌 수 있으면
        D[i] = min(D[i], D[i//3]+1)

print(D[N])

# i를 만드는 카운트를 갱신
```
