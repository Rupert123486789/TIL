T = int(input())

cnt = 0
for _ in range(T):
    H, M = map(int, input().split())
    ans = H * 60 + M - 45
    cnt += 1

    if ans >= 0:
        print(f'Case #{cnt}: {ans//60} {ans%60}')
    else:
        ans += 24*60
        print(f'Case #{cnt}: {ans//60} {ans%60}')
