h1, m1, s1 = map(int, input().split(':'))
h2, m2, s2 = map(int, input().split(':'))

A = h1 * 3600 + m1 * 60 + s1
B = h2 * 3600 + m2 * 60 + s2

if A < B:
    res = B - A
    res_h = res // 3600
    res_m = (res % 3600)//60
    res_s = (res % 3600) % 60
    print(f'{res_h:02d}:{res_m:02d}:{res_s:02d}')
elif A > B:
    res = (86400-A) + B
    res_h = res // 3600
    res_m = (res % 3600)//60
    res_s = (res % 3600) % 60
    print(f'{res_h:02d}:{res_m:02d}:{res_s:02d}')
else:
    print(f'24:00:00')
