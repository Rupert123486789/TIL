def BFS(V):
    que = []
    visited = [0] * (N+1)

    que.append(V)
    visited[V] = 1

    while que:
        S = que.pop(0)
        print(S, end=' ')
        for i in graph[S]:
            if visited[i] == 0:
                que.append(i)
                visited[i] = 1

def DFS(V):
    print(V, end=' ')
    v[V]= 1
    for i in graph[V]:
        if v[i] == 0:
            DFS(i)
            v[i] = 1
